=== SCHEMA DEFINITION ===

# Node Types

## 1) AND/OR Node
{
    "type": "AND" | "OR",
    "children": [<Nodes>, ...]  # At least one child required
}

## 2) COURSE Node
{
    "type": "COURSE",
    "course": str,  # e.g., "MATH 112", "PHYS 121A", "CS 114"
    "min_grade": str | null  # Only: "A", "B+", "B", "C+", "C", "F" - use ONLY when explicitly stated
}

## 3) EQUIVALENT Node  
{
    "type": "EQUIVALENT",
    "courses": [str, ...]  # List of equivalent courses, e.g., ["CS 114", "CS 241"]
}
IMPORTANT: Use this when text says "or equivalent(s)" or lists alternative courses.

## 4) PLACEMENT Node
{
    "type": "PLACEMENT",
    "name": str,  # Raw phrase from input (ALWAYS required)
    "placement_kind": PlacementKind | null,
    "subject": str | null,  # e.g., "MATH", "ENGL", "CHEM"
    "exam": str | null,  # e.g., "ACCUPLACER", "ALEKS", "SAT", "AP"
    "min_course": str | null,  # e.g., "MATH 111" (placement into >= this course)
    "level": str | null,  # e.g., "college algebra", "calculus"
    "min_score": float | str | null  # Use string if mixed formats (e.g., "QAS 250")
}

PlacementKind options:
- "PLACEMENT_INTO_COURSE"
- "PLACEMENT_ABOVE_COURSE"
- "PLACEMENT_TEST_REQUIRED"
- "SCORE_THRESHOLD"
- "DIAGNOSTIC"
- "UNKNOWN"

## 5) PERMISSION Node
{
    "type": "PERMISSION",
    "raw": str,  # Exact text from source (ALWAYS required)
    "permission_kind": PermissionKind | null,
    "authority": PermissionAuthority | null,
    "action": PermissionAction | null,
    "artifact": [str, ...] | null  # Proposal, form, application, etc.
}

PermissionKind options:
- "INSTRUCTOR_APPROVAL"
- "ADVISOR_APPROVAL"
- "DEPARTMENT_APPROVAL"
- "SCHOOL_APPROVAL"
- "PROGRAM_APPROVAL"
- "ADMIN_OVERRIDE"
- "UNKNOWN"

PermissionAuthority options:
- "INSTRUCTOR"
- "FACULTY_SUPERVISOR"
- "DEPARTMENT"
- "SCHOOL"
- "PROGRAM"
- "ADVISOR"
- "REGISTRAR"
- "UNKNOWN"

PermissionAction options:
- "APPROVAL_REQUIRED"
- "SIGNATURE_REQUIRED"
- "PROPOSAL_APPROVAL"
- "APPLICATION_REQUIRED"
- "OVERRIDE_REQUIRED"
- "UNKNOWN"

## 6) STANDING Node
{
    "type": "STANDING",
    "standing": str,  # Raw text from source (ALWAYS required)
    "normalized": str,  # REQUIRED: "FRESHMAN" | "SOPHOMORE" | "JUNIOR" | "SENIOR" | "GRAD"
    "semesters_left": int | null
}

## 7) SKILL Node
{
    "type": "SKILL",
    "name": str  # Exact skill phrase from input
}
IMPORTANT: Only use SKILL for actual skills/knowledge (e.g., "proficiency in Java"). 
DO NOT use for "or equivalent" phrases - those should be EQUIVALENT nodes.

# Course Restrictions

Restriction {
    "raw": str,  # Exact text from catalog/banner (ALWAYS required)
    "kinds": [RestrictionKind, ...] | null,  # Categorization tags

- "INSTRUCTOR_PERMISSION"

# Final Output Format

CourseInfo {
    "prereq_tree": <AndOrNode> | null,
    "coreq_tree": <AndOrNode> | null,
    "restrictions": [<Restriction>, ...]  # Empty array if none
}
